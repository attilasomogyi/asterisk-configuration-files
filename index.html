pjsip.conf

  • pjsip.conf
    • [global]
      • type = global
      • debug = yes
    • [transport-udp]
      • type = transport
      • protocol = udp
      • bind = 0.0.0.0
    • [default-password](!)
      • type = auth
      • auth_type = userpass
      • password = important-change-password-at-first-use
    • [default-from-external](!)
      • type = endpoint
      • context = from-external
      • dtmf_mode = rfc4733
      • disallow = all
      • allow = ulaw
      • direct_media = no
    • [default-from-internal](!)
      • type = endpoint
      • context = from-internal
      • dtmf_mode = rfc4733
      • disallow = all
      • allow = ulaw
      • direct_media = no
    • [default-aors](!)
      • type = aor
      • max_contacts = 1
    • [isdn-1/isdn-2/gsm](default-aors)
    • [isdn-1/isdn-2/gsm](default-password)
      • username = isdn-1/isdn-2/gsm
    • [isdn-1/isdn-2/gsm](default-from-external)
      • auth = isdn-1/isdn-2/gsm
      • aors = isdn-1/isdn-2/gsm
    • [201](default-aors)

      • username = 201
    • [201](default-from-internal)

extensions.conf

  • extensions.conf
    • [general]
      • static=yes
      • wirteprotect=no
      • autofallthrough=yes
    • [globals]
    • [from-external]
      • exten => 201,1,NoOP(${CALLERID})
      • same => n,Ringing
      • same => n,Queue(clients)
      • same => n,Answer
      • same => n,Hangup
    • [from-internal]
      • exten => _0690XXXXXX,1,Hangup/_0681XXXXXX,1,Hangup
      • exten => _06[237]0XXXXXXX,1,Ringing
      • same => 2,Dial(SIP/${EXTEN}@gsm,60)
      • same => 3,GotoIf($[${DIALSTATUS}=CHANUNAVAIL]?:5)
      • same => 4/7/6, Hangup
      • same => 5, Dial(SIP/${EXTEN}@gsm,60)
      • same => 6,GotoIf($[${DIALSTATUS}=CHANUNAVAIL]?:8)
      • same => 8,Dial(SIP/${EXTEN}@isdn-2,60)
      • exten => _06XXXXXXXX,1,Ringing
